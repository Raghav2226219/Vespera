Method: GET
URL: /:boardId
Headers:
  Authorization: Bearer <YOUR_JWT_TOKEN>
Body: None
Example:
GET http://localhost:5896/api/columns/1

Method: PUT
URL: /:boardId/:columnId/rename
Headers:
  Authorization: Bearer <YOUR_JWT_TOKEN>
Body (JSON):
{
  "name": "New Column Name"
}
Example:
PUT http://localhost:5896/api/columns/1/2/rename
Body:
{
  "name": "In Progress"
}

Method: PUT
URL: /:boardId/reorder

  Authorization: Bearer <YOUR_JWT_TOKEN>

{
  "newOrder": [
    { "id": 1, "position": 1 },
    { "id": 2, "position": 2 },
    { "id": 3, "position": 3 }
  ]
}
Example:
PUT http://localhost:5896/api/columns/1/reorder
